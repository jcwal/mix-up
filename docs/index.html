Format for Value & Text Bindings

<p><input data-bind="value: bValue"/></p>
<p><span data-bind="text: bValue"/></p>
<p><input data-bind="checked: bValue" type="checkbox"/></p>
<p><span data-bind="text: bValue, format: true"/></p>
<p><span data-bind="text: bValue" data-format="true"/></p>
<p><span data-bind="text: bValue, format:{'true':'对','false':'错'} "/></p>
<p><span data-bind="text: bValue" data-format='{"true":"Object:真","false":"Object:假"}'/></p>
<p><span data-bind="text: bValue, format: function(x) { return 'Function:'+x;} "/></p>
<p><span data-bind="text: bValue, format: deferredFormat"/></p>
<p><input data-bind="value: bValue, format: deferredFormat"/></p>

<p><input data-bind="value: dValue, format: 'dateTime'"/></p>
<p><span data-bind="text: dValue, format: 'date'"/></p>

<p><span data-bind="text: dValue, format: 'time'"/></p>
<p><span data-bind="text: dValue, format: 'yyyy年MM月dd日 HH:mm:ss'"/></p>
<p><span data-bind="text: dValue" data-format="MM-dd hh:MM:ss"/></p>
<p><span data-bind="text: dValue, format: function(x){ return 'Function:'+ x; }"/></p>

<p><input data-bind="value: sValue"/></p>
<p><span data-bind="text: sValue"/></p>
<p><span data-bind="text: sValue, format: { 'dev':'开发', 'test':'测试', 'prd': '生产' }"/></p>
<p><span data-bind="text: sValue" data-format='{"dev":"开发1", "test":"测试2", "prd": "生产3"}'/></p>
<p><span data-bind="text: 'test,dev'" data-split="true" data-join="^" data-format='{"dev":"开发1", "test":"测试2", "prd": "生产3"}'/></p>
